102 static unsigned char* rle_compress(const unsigned char* data, size_t data_len, size_t* out_len) {
103  if (!data || data_len == 0) return NULL;
104 
105  // Allocate worst case size (input size * 2)
106  unsigned char* compressed = (unsigned char*)malloc(data_len * 2);
107  if (!compressed) return NULL;
108 
109  size_t comp_pos = 0;
110  size_t pos = 0;
111 
112  while (pos < data_len) {
113  unsigned char count = 1;
114  unsigned char current = data[pos];
115 
116  while (pos + count < data_len &&
117  data[pos + count] == current &&
118  count < 255) {
119  count++;
120  }
121 
122  compressed[comp_pos++] = count;
123  compressed[comp_pos++] = current;
124  pos += count;
125  }
126 
127  *out_len = comp_pos;
128  return realloc(compressed, comp_pos); // Shrink to actual size
129 }

173 static unsigned char* lz77_compress(const unsigned char* data, size_t data_len, size_t* out_len) {
174  unsigned char* compressed = (unsigned char*)malloc(data_len * 2);
175  size_t comp_pos = 0;
176 
177  for (size_t i = 0; i < data_len;) {
178  size_t best_len = 0;
179  size_t best_offset = 0;
180 
181  // Search in sliding window
182  size_t window_start = (i > WINDOW_SIZE) ? i - WINDOW_SIZE : 0;
183 
184  for (size_t j = window_start; j < i; j++) {
185  size_t len = 0;
186  while (i + len < data_len &&
187  j + len < i &&
188  data[i + len] == data[j + len] &&
189  len < 255) {
190  len++;
191  }
192 
193  if (len > best_len) {
194  best_len = len;
195  best_offset = i - j;
196  }
197  }
198 
199  if (best_len >= MIN_MATCH) {
200  compressed[comp_pos++] = 1; // Flag for match
201  compressed[comp_pos++] = best_offset & 0xFF;
202  compressed[comp_pos++] = (best_offset >> 8) & 0xFF;
203  compressed[comp_pos++] = best_len;
204  i += best_len;
205  } else {
206  compressed[comp_pos++] = 0; // Flag for literal
207  compressed[comp_pos++] = data[i++];
208  }
209  }
210 
211  *out_len = comp_pos;
212  return realloc(compressed, comp_pos);
213 }

223 static unsigned char* lz77_decompress(const unsigned char* compressed, size_t comp_len, size_t* out_len) {
224  if (!compressed || comp_len == 0) return NULL;
225 
226  unsigned char* decompressed = (unsigned char*)malloc(comp_len * 2); // Initial size estimate
227  size_t decom_pos = 0;
228  size_t pos = 0;
229 
230  while (pos < comp_len) {
231  if (compressed[pos] == 0) { // Literal
232  decompressed[decom_pos++] = compressed[pos + 1];
233  pos += 2;
234  } else { // Match
235  size_t offset = compressed[pos + 1] | (compressed[pos + 2] << 8);
236  size_t length = compressed[pos + 3];
237 
238  for (size_t i = 0; i < length; i++) {
239  decompressed[decom_pos] = decompressed[decom_pos - offset];
240  decom_pos++;
241  }
242  pos += 4;
243  }
244  }
245 
246  *out_len = decom_pos;
247  return realloc(decompressed, decom_pos);

583

585  SDBEntry* entry = t->entries->entries[hash];
586 
587  // Handle collision with linear probing

591  }

600  } else {
601  t->entries->tail->next = e;
602  }
